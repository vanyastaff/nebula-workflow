1use axum::{
2    body::{self, Body},
3    extract::State,
4    http::{Request, Response, StatusCode},
5    middleware::Next,
6};
7use std::collections::HashMap;
8use std::sync::{Arc, Mutex};
9
10pub type IdempotencyCache = Arc<Mutex<HashMap<String, Vec<u8>>>>;
11
12pub async fn idempotency_middleware(
13    State(cache): State<IdempotencyCache>,
14    req: Request<Body>,
15    next: Next,
16) -> Response<Body> {
17    if req.method() != axum::http::Method::POST {
18        return next.run(req).await;
19    }
20    let key = req
21        .headers()
22        .get("Idempotency-Key")
23        .and_then(|v| v.to_str().ok())
24        .map(|s| s.to_string());
25    if let Some(key) = key {
26        if let Some(cached) = cache.lock().unwrap().get(&key).cloned() {
27            return Response::builder()
28                .status(StatusCode::OK)
29                .body(Body::from(cached))
30                .unwrap();
31        }
32        let response = next.run(req).await;
33        let (parts, body) = response.into_parts();
34        let body_bytes = body::to_bytes(body, 2 * 1024 * 1024).await.unwrap();
35        cache.lock().unwrap().insert(key, body_bytes.clone().to_vec());
36        Response::from_parts(parts, Body::from(body_bytes))
37    } else {
38        Response::builder()
39            .status(StatusCode::BAD_REQUEST)
40            .body(Body::from("Missing Idempotency-Key header"))
41            .unwrap()
42    }
43}
